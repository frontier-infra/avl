# AVL Manifest
# Discovery file for AI agents. Like robots.txt but for agent-native views.

version: 1
content-type: text/agent-view; version=1

# Discovery mechanisms supported
discovery:
  - suffix          # GET /<route>.agent
  - accept-header   # Accept: text/agent-view
  - native          # GET /agent/<route>

# Auth — the agent inherits the host app's session (delegate, not principal)
session:
  mechanisms: [cookie, bearer]
  cookie: session_token
  bearer_prefix: Bearer
  scope: per-user

# Routes (pattern, method)
routes:
  - GET /dashboard.agent
  - GET /journey/{id}.agent

spec: https://github.com/frontier-infra/avl
